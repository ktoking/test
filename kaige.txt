#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

void fun(int a[], int n, int l) {
	for (int i = 0; i < n; i++)
	{
		printf("%d", a[i]);
	}

}


int main()
{
	int n, l;
	int a[100];
	int flag = 0;
	while (scanf("%d %d", &n, &l) != EOF) {
		if (n >= 2 && n <= 100000 && l >= 0 && l <= 10000) {
			for (int i = 0; i < n; i++)
			{
				scanf("%d", &a[i]);
				if (a[i] < 0 || a[i] > 10000) {
					flag = 1;
					break;
				}
			}
			if (flag == 1) {
				break;
			}
			fun(a, n, l);
		}
	}
	system("pause");
}
